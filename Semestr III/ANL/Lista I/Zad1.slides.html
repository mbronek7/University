require 'nyaplot'
class Function
 
  def initialize(&block)
     @function = block
  end

  def value (x)   
    @function.call x
  end
 
  
 def plot a, b
   x = []; y = []; 
   x = (a..b) .collect { |v| v.to_f }
   y = x.collect { |v| value(v)}
   plot1 = Nyaplot::Plot.new
   plot1.add(:line, x, y)
   plot1.show
end

def zero_of_fun(a,b,c)
   begin 
   delta = b*b-4*a*c 
   x1 = (-b + Math.sqrt(delta)) / (2.0*a)
   x2 = (-b - Math.sqrt(delta)) / (2.0*a)
   print "Miejsca zerowe to: x1 = #{x1} || x2 = #{x2}\n"
   print"Wartość funkcji w tych punktach to odpowiednio dla #{x1} -- #{@function.call x1}, a dla #{x2} --#{@function.call x2}\n "
   rescue 
   print "delta < 0"
   end
end
  
end
